CREATE TABLE Videos(Titre varchar(100) primary KEY,Description TEXT,Duree time,Date_Parution date,Lien_Affiche varchar(170))

CREATE TABLE Videos_Films(Titre varchar(100) PRIMARY KEY,foreign KEY(Titre) REFERENCES Videos(Titre) On DELETE CASCADE)

CREATE TABLE Videos_Series (
	Titre	varchar(100),
	PRIMARY KEY(Titre),
	FOREIGN KEY(Titre) REFERENCES Videos(Titre)  On DELETE CASCADE
)

CREATE TABLE Details_Videos_Series(Titre varchar(100),Saison INTEGER,Episode INTEGER,PRIMARY KEY(Titre,Saison,Episode),FOREIGN KEY(Titre) REFERENCES Videos_Series(Titre) On DELETE CASCADE)

CREATE TABLE Animes_Series(Titre varchar(100) primary KEY,Studio varchar(30),foreign KEY(Titre) REFERENCES Videos_Series(Titre) On DELETE CASCADE)

CREATE TABLE Animes_Films (
	Titre	varchar(100),
	Studio	varchar(30),
	FOREIGN KEY(Titre) REFERENCES Videos_Films(Titre) On DELETE CASCADE,
	PRIMARY KEY(Titre)
)

CREATE TABLE Dramas_Films (
	Titre	varchar(100),
	PRIMARY KEY(Titre),
	FOREIGN KEY(Titre) REFERENCES Videos_Films(Titre)  On DELETE CASCADE
)

CREATE TABLE Dramas_Series(Titre varchar(100) PRIMARY KEY,FOREIGN KEY(Titre) REFERENCES Videos_Series(Titre) On DELETE CASCADE)

CREATE TABLE Films(Titre varchar(100) primary KEY,FOREIGN KEY(Titre) REFERENCES Videos_Films(Titre) On DELETE CASCADE)

CREATE TABLE Series(Titre varchar(100) primary KEY,foreign KEY(Titre) REFERENCES Videos_Series(Titre) On DELETE CASCADE)

CREATE TABLE Genres(Nom varchar(50) primary KEY)

CREATE TABLE Pays(Nom varchar(50) primary KEY)

CREATE TABLE Realisateurs(Nom varchar(50) primary KEY)

CREATE TABLE Videos_Genres(Titre varchar(100),Nom varchar(50),primary KEY(Titre,Nom),foreign KEY(Titre) REFERENCES Videos(Titre) On DELETE CASCADE,FOREIGN KEY(Nom) REFERENCES Genres(Nom) On DELETE CASCADE)

CREATE TABLE Videos_Pays(Titre varchar(100),Nom varchar(50),primary KEY(Titre,Nom),FOREIGN KEY(Titre) REFERENCES Videos(Titre) On DELETE CASCADE,FOREIGN KEY(Nom) REFERENCES Pays(Nom) On DELETE CASCADE)

CREATE TABLE Videos_Realisateurs(Titre varchar(100),Nom varchar(50),primary KEY(Titre,Nom),foreign KEY(Titre) REFERENCES Videos(Titre) On DELETE CASCADE,FOREIGN KEY(Nom) REFERENCES Realisateurs(Nom) On DELETE CASCADE)
